`SequenceMetaForm` adapts to `type="broadcast"` or `type="sequence"`.

## Usage

This is the exact code behind the demos above.

```tsx
import { useState } from "react";
import {
  SequenceMetaForm,
  type ContactFilterWithAggregator,
  type SequenceMetaFormValue,
} from "@sendlit/email-blocks";

const initialFilter: ContactFilterWithAggregator = {
  aggregator: "and",
  filters: [
    { name: "tag", condition: "is", value: "customers" },
    { name: "email", condition: "contains", value: "@example.com" },
  ],
};

function BroadcastExample() {
  const [filter, setFilter] =
    useState<ContactFilterWithAggregator>(initialFilter);
  const [broadcastMeta, setBroadcastMeta] = useState<SequenceMetaFormValue>({
    title: "July product announcement",
    fromName: "SendLit Team",
    fromEmail: "hello@sendlit.dev",
    filter,
  });

  return (
    <SequenceMetaForm
      type="broadcast"
      value={{ ...broadcastMeta, filter }}
      onChange={(value) => {
        setBroadcastMeta(value);
        if (value.filter) setFilter(value.filter);
      }}
    />
  );
}

function SequenceExample() {
  const [sequenceMeta, setSequenceMeta] = useState<SequenceMetaFormValue>({
    title: "New subscriber onboarding",
    fromName: "SendLit Team",
    fromEmail: "hello@sendlit.dev",
    triggerType: "tag:added",
    triggerData: "trial",
  });

  return (
    <SequenceMetaForm
      type="sequence"
      value={sequenceMeta}
      onChange={setSequenceMeta}
    />
  );
}
```
