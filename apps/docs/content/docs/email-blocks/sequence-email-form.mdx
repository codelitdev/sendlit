`SequenceEmailForm` wraps subject, publish state, delay/action controls, and the full email editor.

<EmailBlockDemo demo="sequence-email-form" />

## Usage

This is the exact code behind the demo above (with `content` shortened to `defaultTemplateEmail`
here for brevity — the demo uses a fuller sample email).

```tsx
import { useState } from "react";
import {
  SequenceEmailForm,
  defaultTemplateEmail,
  type SequenceEmailFormValue,
} from "@sendlit/email-blocks";

const day = 86_400_000;

function Example() {
  const [sequenceEmail, setSequenceEmail] = useState<SequenceEmailFormValue>({
    subject: "Make your next announcement premium",
    content: defaultTemplateEmail,
    delayInMillis: day,
    published: true,
    actionType: "tag:add",
    actionData: { tag: "engaged" },
  });

  return (
    <SequenceEmailForm
      value={sequenceEmail}
      onChange={setSequenceEmail}
      variant="sequence"
      className="h-full"
    />
  );
}
```
